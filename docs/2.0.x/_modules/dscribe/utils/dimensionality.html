[docs]def is1d(array, dtype=np.integer):
    try:
        for i in array:
            if not np.issubdtype(type(i), dtype):
                return False
    except Exception:
        return False
    return True




[docs]def is2d(array, dtype=np.integer):
    try:
        for i in array:
            for j in i:
                if not np.issubdtype(type(j), dtype):
                    return False
    except Exception:
        return False
    return True
